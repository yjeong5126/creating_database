-- create table Orders

CREATE TABLE Orders (
    OrderID INT NOT NULL AUTO_INCREMENT,
    OrderNumber INT(5),
    OrderDate DATE,
    Quarter INT,
    Month INT,
    Year INT,
    StatusID INT,
    CustomerID INT,
    ContactNameID INT,
    PRIMARY KEY (OrderID),
    FOREIGN KEY (StatusID)
        REFERENCES Status (StatusID)
        ON DELETE CASCADE ON UPDATE CASCADE,
    FOREIGN KEY (CustomerID)
        REFERENCES Customers (CustomerID)
        ON DELETE CASCADE ON UPDATE CASCADE,
    FOREIGN KEY (ContactNameID)
        REFERENCES ContactNames (ContactNameID)
        ON DELETE CASCADE ON UPDATE CASCADE
);
